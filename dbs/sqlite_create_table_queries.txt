CREATE TABLE "admin_details" ( "USERNAME" TEXT NOT
NULL, "PASSWORD" TEXT NOT NULL, "FULL_NAME" TEXT
NOT NULL, "DOB" TEXT NOT NULL, "SEX" TEXT NOT
NULL, "PHONE" TEXT NOT NULL, "EMAIL" TEXT NOT NULL
);

CREATE TABLE "employees" ( "EMPLOYEE_ID" TEXT NOT
NULL UNIQUE, "NAME" TEXT NOT NULL, "DOB" TEXT NOT
NULL, "SEX" TEXT NOT NULL, "DEPARTMENT" TEXT NOT
NULL, "PHONE" TEXT NOT NULL, "EMAIL" TEXT NOT
NULL, "GROSS_SALARY" REAL NOT NULL,
"TOTAL_ALLOWANCE" REAL NOT NULL DEFAULT 0.00,
PRIMARY KEY("EMPLOYEE_ID") );

CREATE TABLE "monthly_income_tax_rates" (
"CHARGEABLE_INCOME" TEXT NOT NULL, "RATE" REAL NOT
NULL DEFAULT 0, "TAX" REAL NOT NULL DEFAULT 0,
"CUMMULATIVE_CHARGEABLE_INCOME" REAL NOT NULL,
"CUMMULATIVE_TAX" REAL NOT NULL DEFAULT 0,
"ROW_ID" INTEGER NOT NULL );

CREATE TABLE "payment_receipts" ( "PAYMENT_ID"
INTEGER, "TRANSACTION_TIME" TEXT NOT NULL,
"RECIPIENTS" TEXT NOT NULL, "TOTAL_AMOUNT" REAL
NOT NULL, PRIMARY KEY("PAYMENT_ID" AUTOINCREMENT)
)
